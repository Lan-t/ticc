stmt    :   assign ";"
        :   "return" assign ";"

assign  :   add
        |   add "=" assign

add     :   mul
        |   add "+" mul
        |   add "-" mul

mul     :   term
        |   mul "*" term
        |   mul "/" term

term    :   <num>
        |   <ident>
        |   "(" assign ")"

<ident> :   'a' <= c <= 'z';
